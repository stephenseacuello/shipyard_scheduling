\section{Method}
%==============================================================

\begin{frame}{MDP Formulation}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{State} (heterogeneous graph)
\begin{itemize}
    \item Block nodes: stage, progress, deadline
    \item Equipment nodes: health, location, load
    \item Ship nodes: completion \%, dock assignment
    \item Edge types: assigned-to, queued-at, depends-on
\end{itemize}

\vspace{0.3em}
\textbf{GNN Encoder}
\begin{itemize}
    \item 3-layer GATv2 with edge features
    \item Global attention pooling
    \item Fixed-size embedding ($\mathbb{R}^{256}$)
\end{itemize}

\column{0.48\textwidth}
\textbf{Action} (hierarchical, 10 heads)
\begin{itemize}
    \item Action type: transport / erect / hold / maintain
    \item Equipment: which SPMT or crane
    \item Target: which block to move
    \item Only relevant heads contribute to $\log\pi$
\end{itemize}

\vspace{0.3em}
\textbf{Reward}
\begin{itemize}
    \item $+1$ per block stage advance
    \item $+10$ ship delivery bonus
    \item $-0.5$ deadline violation penalty
    \item Equipment health penalties
\end{itemize}
\end{columns}
\end{frame}

%--------------------------------------------------------------
\begin{frame}{Training: Why DAgger over PPO?}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{PPO Entropy Collapse}
\includegraphics[width=\textwidth]{figures/entropy_collapse.png}

\begin{itemize}
    \item Entropy: $2.27 \to 0.00$ by epoch 5
    \item Policy becomes deterministic too fast
    \item Hierarchical action space amplifies issue
\end{itemize}

\column{0.48\textwidth}
\textbf{DAgger with Curriculum}
\includegraphics[width=\textwidth]{figures/curriculum_dagger.png}

\begin{itemize}
    \item Expert: EDD (Earliest Due Date) dispatch
    \item 3-stage curriculum: 10 $\to$ 50 $\to$ 200 blocks
    \item Observation normalization (Welford's)
    \item Per-stage beta reset: $0.8 \to 0.2$
\end{itemize}
\end{columns}
\end{frame}

\section{Key Findings \& Limitations}
%==============================================================

\begin{frame}{Key Findings}
\begin{enumerate}
    \item \textbf{Scalability is the bottleneck}: All methods solve small instances; the challenge is scaling to 200+ blocks
    \item \textbf{DAgger $>$ PPO for hierarchical actions}: Imitation learning avoids entropy collapse entirely
    \item \textbf{Observation normalization matters}: RunningMeanStd normalizer must be saved with checkpoint --- different statistics at deployment cause complete failure
    \item \textbf{Curriculum shows transient learning}: Peak 34\% during training on medium, but final checkpoint regresses to 0\%
    \item \textbf{Expert (EDD) is remarkably strong}: Simple priority dispatch scales better than learned policies and optimization methods
    \item \textbf{OR baselines don't scale either}: GA and MPC hit computational limits before RL hits learning limits
\end{enumerate}
\end{frame}

    \item Real-world validation at Korean shipyards
    \item Weather-dependent scheduling
